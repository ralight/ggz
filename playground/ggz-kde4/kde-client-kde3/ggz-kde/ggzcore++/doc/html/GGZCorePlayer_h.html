#ifndef GGZPLAYER_H
#define GGZPLAYER_H

#include 

/**
* Player representation.
*/

class GGZCorePlayer
{
	public:
		/**
		 * Constructor */
		GGZCorePlayer();
		/**
		 * Destructor */
		~GGZCorePlayer();

		/**
		 * Available player types */
		enum PlayerTypes
		{
			none,
			normal,
			guest,
			admin
		};

		/**
		 * Return the name of this player. */
		char* name();
		/**
		 * Return the player's current table. */
		GGZTable* table();
		/**
		 * Return the player's type */
		int type();

		/**
		 * Lag measurement */
		int lag();

	private:
		GGZPlayer *m_player;
};

#endif
